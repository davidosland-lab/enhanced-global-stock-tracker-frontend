def sentiment_prediction(self, sentiment_data, current_price):
    """
    Generate prediction based on FinBERT sentiment analysis
    
    Args:
        sentiment_data: Dict with keys 'compound', 'confidence', 'sentiment'
        current_price: Current stock price
    
    Returns:
        Dict with prediction, predicted_price, confidence, model_type
    """
    if not sentiment_data or not isinstance(sentiment_data, dict):
        return None
    
    compound = sentiment_data.get('compound', 0)
    confidence = sentiment_data.get('confidence', 50)
    
    # Strong positive sentiment â†’ BUY
    if compound > 0.3:
        prediction = 'BUY'
        predicted_change = 2.0  # Expect 2% increase
        
    # Strong negative sentiment â†’ SELL
    elif compound < -0.3:
        prediction = 'SELL'
        predicted_change = -2.0  # Expect 2% decrease
        
    # Neutral sentiment â†’ HOLD
    else:
        prediction = 'HOLD'
        predicted_change = 0.3  # Small positive drift
    
    predicted_price = current_price * (1 + predicted_change / 100)
    
    # Sentiment confidence is usually high, so cap it
    adjusted_confidence = min(confidence, 85)
    
    return {
        'prediction': prediction,
        'predicted_price': predicted_price,
        'confidence': adjusted_confidence,
        'model_type': 'FinBERT Sentiment',
        'reasoning': f'News sentiment: {sentiment_data.get("sentiment", "neutral")}, compound score: {compound:.2f}'
    }

def technical_indicators_prediction(self, stock_data, current_price):
    """
    Enhanced technical analysis with multiple indicators
    Uses 'ta' library for advanced indicators
    """
    df = stock_data.copy()
    
    signals = []  # List of BUY/SELL/HOLD signals
    
    # 1. Simple Moving Averages (SMA)
    df['SMA_20'] = ta.trend.sma_indicator(df['Close'], window=20)
    df['SMA_50'] = ta.trend.sma_indicator(df['Close'], window=50)
    df['SMA_200'] = ta.trend.sma_indicator(df['Close'], window=200)
    
    if not df['SMA_50'].isna().all():
        if current_price > df['SMA_50'].iloc[-1]:
            signals.append('BUY')
        else:
            signals.append('SELL')
    
    if not df['SMA_200'].isna().all():
        # Golden Cross: SMA 50 > SMA 200 (bullish)
        if df['SMA_50'].iloc[-1] > df['SMA_200'].iloc[-1]:
            signals.append('BUY')
        # Death Cross: SMA 50 < SMA 200 (bearish)
        else:
            signals.append('SELL')
    
    # 2. RSI (Relative Strength Index)
    df['RSI'] = ta.momentum.rsi(df['Close'], window=14)
    
    if not df['RSI'].isna().all():
        rsi_value = df['RSI'].iloc[-1]
        if rsi_value < 30:  # Oversold â†’ BUY
            signals.append('BUY')
        elif rsi_value > 70:  # Overbought â†’ SELL
            signals.append('SELL')
        else:
            signals.append('HOLD')
    
    # 3. MACD (Moving Average Convergence Divergence)
    macd = ta.trend.MACD(df['Close'])
    df['MACD'] = macd.macd()
    df['MACD_signal'] = macd.macd_signal()
    
    if not df['MACD'].isna().all():
        if df['MACD'].iloc[-1] > df['MACD_signal'].iloc[-1]:
            signals.append('BUY')
        else:
            signals.append('SELL')
    
    # 4. Bollinger Bands
    bb = ta.volatility.BollingerBands(df['Close'])
    df['BB_upper'] = bb.bollinger_hband()
    df['BB_lower'] = bb.bollinger_lband()
    
    if not df['BB_upper'].isna().all():
        if current_price > df['BB_upper'].iloc[-1]:
            signals.append('SELL')  # Overbought
        elif current_price < df['BB_lower'].iloc[-1]:
            signals.append('BUY')  # Oversold
        else:
            signals.append('HOLD')
    
    # 5. Stochastic Oscillator
    stoch = ta.momentum.StochasticOscillator(df['High'], df['Low'], df['Close'])
    df['Stoch'] = stoch.stoch()
    
    if not df['Stoch'].isna().all():
        stoch_value = df['Stoch'].iloc[-1]
        if stoch_value < 20:  # Oversold
            signals.append('BUY')
        elif stoch_value > 80:  # Overbought
            signals.append('SELL')
        else:
            signals.append('HOLD')
    
    # 6. EMA (Exponential Moving Averages)
    df['EMA_12'] = ta.trend.ema_indicator(df['Close'], window=12)
    df['EMA_26'] = ta.trend.ema_indicator(df['Close'], window=26)
    
    if not df['EMA_12'].isna().all():
        if df['EMA_12'].iloc[-1] > df['EMA_26'].iloc[-1]:
            signals.append('BUY')
        else:
            signals.append('SELL')
    
    # Count signals
    buy_signals = signals.count('BUY')
    sell_signals = signals.count('SELL')
    hold_signals = signals.count('HOLD')
    total_signals = len(signals)
    
    # Determine consensus
    if buy_signals > sell_signals and buy_signals > hold_signals:
        prediction = 'BUY'
        predicted_change = 1.5
    elif sell_signals > buy_signals and sell_signals > hold_signals:
        prediction = 'SELL'
        predicted_change = -1.5
    else:
        prediction = 'HOLD'
        predicted_change = 0.3
    
    # Calculate confidence based on consensus
    max_signals = max(buy_signals, sell_signals, hold_signals)
    confidence = 50 + (max_signals / total_signals) * 40  # 50-90%
    
    predicted_price = current_price * (1 + predicted_change / 100)
    
    return {
        'prediction': prediction,
        'predicted_price': predicted_price,
        'confidence': confidence,
        'model_type': 'Technical Indicators (Enhanced)',
        'reasoning': f'{buy_signals} BUY, {sell_signals} SELL, {hold_signals} HOLD signals from {total_signals} indicators'
    }

Run this overnight to train models for most-traded stocks

from datetime import datetime

def train_lstm_for_stock(predictor, symbol):
    """Train and save LSTM model for a single stock"""
    print(f"\n{'='*60}")
    print(f"Training LSTM for {symbol}")
    print(f"{'='*60}")
    
    try:
        # Download 2 years of data
        print(f"Downloading {symbol} data (2 years)...")
        ticker = yf.Ticker(symbol)
        data = ticker.history(period='2y')
        
        if len(data) < 100:
            print(f"âŒ {symbol}: Not enough data ({len(data)} days)")
            return False
        
        print(f"âœ“ Downloaded {len(data)} days of data")
        
        # Train LSTM
        print(f"Training LSTM (this may take 5-10 minutes)...")
        predictor.train_and_save_lstm(data, symbol)
        
        print(f"âœ“ {symbol}: Training complete!")
        return True
        
    except Exception as e:
        print(f"âŒ {symbol}: Training failed - {str(e)}")
        return False
