**Integration Code:**
```python
def _get_spi_from_commsec_export(self) -> Dict:
    """
    Read SPI 200 futures from CommSec CSV export
    """
    try:
        # Read CSV export
        csv_path = Path("data/commsec_data.csv")
        
        if not csv_path.exists():
            logger.warning("CommSec export not found")
            return {'available': False}
        
        # Check file age (should be < 5 minutes old)
        file_age = time.time() - csv_path.stat().st_mtime
        if file_age > 300:  # 5 minutes
            logger.warning(f"CommSec data stale ({file_age/60:.1f} min old)")
            return {'available': False}
        
        # Parse CSV
        df = pd.read_csv(csv_path)
        
        # Find SPI 200 row
        spi_row = df[df['Code'] == 'AP']
        
        if spi_row.empty:
            logger.warning("SPI 200 (AP) not found in CommSec export")
            return {'available': False}
        
        spi_price = float(spi_row['Last'].iloc[0])
        
        # Get ASX 200 close
        axjo_row = df[df['Code'] == 'XJO']
        asx_close = float(axjo_row['Last'].iloc[0])
        
        # Calculate gap
        gap_pct = ((spi_price - asx_close) / asx_close) * 100
        
        return {
            'available': True,
            'spi_price': spi_price,
            'asx_close': asx_close,
            'gap_pct': gap_pct,
            'timestamp': csv_path.stat().st_mtime,
            'source': 'COMMSEC_EXPORT'
        }
        
    except Exception as e:
        logger.error(f"Failed to read CommSec export: {e}")
        return {'available': False}
```

```python
from selenium import webdriver
from selenium.webdriver.common.by import By

def login_commsec(username, password):
    driver = webdriver.Chrome()
    driver.get("https://www2.commsec.com.au/secure/login")
    
    # Login
    driver.find_element(By.ID, "username").send_keys(username)
    driver.find_element(By.ID, "password").send_keys(password)
    driver.find_element(By.ID, "login").click()
    
    return driver

def get_spi_price(driver):
    # Navigate to SPI 200 quote
    driver.get("https://www2.commsec.com.au/quote?code=AP")
    
    # Extract price
    price_element = driver.find_element(By.CLASS_NAME, "last-price")
    spi_price = float(price_element.text.replace(',', ''))
    
    return spi_price
```
